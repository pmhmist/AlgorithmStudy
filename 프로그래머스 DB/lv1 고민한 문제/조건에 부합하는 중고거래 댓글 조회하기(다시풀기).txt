SELECT A.TITLE, A.BOARD_ID, 
    B.REPLY_ID, B.WRITER_ID, B.CONTENTS, TO_CHAR(B.CREATED_DATE,'YYYY-MM-DD') AS CREATED_DATE
FROM USED_GOODS_BOARD A JOIN USED_GOODS_REPLY B
ON A.BOARD_ID = B.BOARD_ID
WHERE A.CREATED_DATE BETWEEN TO_DATE('2022-10-01','YYYY-MM-DD') 
    AND TO_DATE('2022-10-31','YYYY-MM-DD') + 0.99999
ORDER BY B.CREATED_DATE, A.TITLE;
