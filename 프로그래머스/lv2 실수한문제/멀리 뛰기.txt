class Solution {
    public long solution(int n) {
        long answer = 0;
        long mod = 1234567;
        long[] arr = new long[n+2];
        arr[1] = 1;
        arr[2] = 2;
        for(int i = 3; i <= n; i++){
            arr[i] = (arr[i-1] + arr[i-2])%mod;
        }
        answer = arr[n];
        
        return answer;
    }
}
